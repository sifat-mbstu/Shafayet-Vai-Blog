int v,w;

void mst(int src,int n)
{
priority_queuepq;
int visited[2000]={0},i,u,v,sum=0,j,p;
for(i=1;i<n;i++)
{
u=src;
visited[src]=1;
for(j=0;j<vec[u].size();j++)
{
v=vec[u][j];
if(visited[v]==0)
{
data D; D.v=v;
D.w=cost[u][j];
pq.push(D);
}
}
do{
data T=pq.top();
src=T.v;
p=T.w;
pq.pop();}while(visited[src]==1);
sum+=p;
}
printf("%d",sum);
}
